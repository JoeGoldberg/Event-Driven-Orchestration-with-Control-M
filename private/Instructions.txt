# Installing Kafka
tar -xzf kafka_2.13-3.6.1.tgz
cd kafka_2.13-3.6.1

# Start the ZooKeeper service
bin/zookeeper-server-start.sh config/zookeeper.properties

# In a different terminal session, start the Kafka broker service
bin/kafka-server-start.sh config/server.properties

# Create a topic
bin/kafka-topics.sh --create --topic quickstart-events --bootstrap-server localhost:9092

# Describe newly-created topic
bin/kafka-topics.sh --describe --topic quickstart-events --bootstrap-server localhost:9092

# Produce (Write) to topic
bin/kafka-console-producer.sh --topic ctm-events --bootstrap-server localhost:9092

# Consume (read) from topic
bin/kafka-console-consumer.sh --topic ctm-events --from-beginning --bootstrap-server localhost:9092

# Kafka Connect data generator
https://github.com/xushiyan/kafka-connect-datagen


consumer = KafkaConsumer('ctm-events', bootstrap_servers=['localhost:9092'])
